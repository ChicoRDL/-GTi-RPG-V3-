GUIEditor = {
    staticimage = {},
    scrollbar = {},
    label = {}
}

    function()

        GUIEditor.scrollbar[1] = guiCreateScrollBar(251, -4, 22, 397, false, false, GUIEditor.staticimage[1])
        guiScrollBarSetScrollPosition(GUIEditor.scrollbar[1], 100.0)


        GUIEditor.staticimage[2] = guiCreateStaticImage(599, 224, 270, 36, ":GTIMessagingApp/images/bckgrnd.png", false)

        GUIEditor.staticimage[3] = guiCreateStaticImage(231, 6, 25, 25, ":GTIMessagingApp/images/friend.png", false, GUIEditor.staticimage[2])
        GUIEditor.staticimage[4] = guiCreateStaticImage(7, 5, 33, 26, ":GTIMessagingApp/images/msg.png", false, GUIEditor.staticimage[2])
        GUIEditor.label[3] = guiCreateLabel(50, 10, 158, 16, "GTIdroid SMS Application", false, GUIEditor.staticimage[2])

        guiSetFont(GUIEditor.label[3], font_0)
        guiLabelSetColor(GUIEditor.label[3], 195, 248, 255)    
    end
)
